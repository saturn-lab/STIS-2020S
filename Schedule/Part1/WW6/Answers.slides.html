def ALLEQ_N(a, b, c, d):
    p1 = NAND(a,b)
    p2 = NAND(c,d)
    np1 = NAND(p1, p1)
    np2 = NAND(p2, p2)
    r1 = NAND(np1, np2)
    na = NAND(a,a)
    nb = NAND(b,b)
    nc = NAND(c,c)
    nd = NAND(d,d)
    q1 = NAND(na,nb)
    q2 = NAND(nc,nd)
    nq1 = NAND(q1, q1)
    nq2 = NAND(q2, q2)
    r2 = NAND(nq1, nq2)
    return NAND(r1, r2)

def ALLEQ5(a, b, c, d, e):
    r1 = AND(AND(AND(a,b),AND(c,d)), e)
    r2 = NOT(OR(OR(OR(a,b),OR(c,d)),e))
    return OR(r1, r2)

def SEL(a, b, c):
    r1 = AND(a, b)
    r2 = AND(NOT(a), c)
    return OR(r1, r2)

class MAJ(nn.Module):
    def __init__(self):
        super().__init__()
        self.ninput = 3
        self.and_ = AND()
        self.or_ = OR()

    def forward(self, x):
        t1 = self.and_(zt(x[0], x[1]))
        t2 = self.and_(zt(x[1], x[2]))
        t3 = self.and_(zt(x[0], x[2]))
        y1 = self.or_(zt(t2,t3))
        return self.or_(zt(y1, t1))
