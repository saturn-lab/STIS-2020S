    return NAND(NAND(NAND(NAND(a,b),NAND(a,c)),NAND(NAND(a,b),NAND(a,c))),NAND(b,c))

def MAJ2(a,b,c):
    return OR(AND(a,b), OR(AND(b,c), AND(a,c)))

def CMP(x0, x1, x2, x3):
    u = AND(NOT(x2), x0)
    v = OR(x0, x2)
    w = AND(x1, NOT(x3))
    return OR(AND(u,v), w)

def ALLEQ(a, b, c, d):
    u = XOR(a, b)
    v = XOR(c, d)
    w = XOR(b, c)
    return NOT(OR(u, OR(v, w)))
